Options
  • position
    • type: String
    • default: top
    • description: top, bottom, middle 중 선택
  • top
    • type: Number
    • default: 0
    • description: sticky 활성화 위치
  • activeClassName
    • type: String
    • default: is-sticky
    • description: sticky 활성화 클래스

- Event

JAVASCRIPT 

  var yourVariable = $('.your-element');

  yourVariable.on('beforeChange', function( event, plugin, anchor, panel ){

  yourVariable.on('afterChange', function( event, plugin, anchor, panel ){

  • beforeChange
    • arguments: event, plugin, anchor, panel
    • description: 스티키 활성 전 실행될 함수
  • afterChange
    • arguments: event, plugin, anchor, panel
    • description: 스티키 활성 후 실행될 함수
